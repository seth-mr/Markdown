int cantidadANumero(string cantidadC){
	string cadNueva = "";
	int i = 0;
	int cantidadNumero;

	while(cantidadC[i] != '\0'){
		if(cantidadC[i] > 44 && cantidadC[i] < 58){
			cadNueva = cadNueva + cantidadC[i];
		}

		i++;
	}

	if(cadNueva == ""){
		cantidadNumero = -1;
	}
	else{
		
		cantidadNumero = stoi(cadNueva);
	}


	return cantidadNumero;
}
